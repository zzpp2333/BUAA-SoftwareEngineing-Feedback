int mod(int a,int b){
	int c;
	c = a / b;
	b = b * c;
	if(a == b)
		return(0);
	return(a-b);
}

int gcd(int a,int b){
	int c;
	if(a < b){
		c = a;
		a = b;
		b = c;
	}
	if(mod(a,b) == 0){
		return(b);
	}
	return(gcd(a,mod(a,b)));
}
